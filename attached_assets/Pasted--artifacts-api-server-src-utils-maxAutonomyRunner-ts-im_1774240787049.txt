import { executeTask } from "./taskExecutor"; // existing single-task executor
import { db } from "../db";
import { logInfo, logError } from "./logger";

// Type for a task
interface Task {
  name: string;
  run: () => Promise<any>;
  safe: boolean; // must pass safety/compliance checks
}

// Max-Autonomy Runner
export async function runMaxAutonomy(tasks: Task[], inventNewTasks: boolean = true) {
  try {
    // 1️⃣ Filter for safe tasks only
    const safeTasks = tasks.filter((t) => t.safe);

    // 2️⃣ Run all safe tasks in parallel with minimal token use
    const taskPromises = safeTasks.map((t) =>
      executeTask(t.name, t.run).catch((err) => {
        logError(`Task failed: ${t.name}`, err);
        return null; // fail-safe
      })
    );

    // 3️⃣ Wait for all existing tasks
    const results = await Promise.all(taskPromises);
    logInfo(`Completed ${safeTasks.length} tasks successfully.`);

    // 4️⃣ Invent new tasks if enabled
    if (inventNewTasks) {
      const inventedTasks: Task[] = await discoverNewSafeTasks(results);
      if (inventedTasks.length > 0) {
        logInfo(`Discovered ${inventedTasks.length} new tasks.`);
        // Run invented tasks recursively (safe + minimal tokens)
        const inventedResults = await runMaxAutonomy(inventedTasks, false);
        return { results, inventedResults };
      }
    }

    return { results };
  } catch (err) {
    logError("Max Autonomy Runner failed unexpectedly", err);
    return { results: [], error: err };
  }
}

  const newTasks: Task[] = [];

  // Example patterns: analytics expansions, notification optimizations, revenue insights
  // Only create tasks that are safe/legal

      name: "PredictiveChurnAlerts",

}
