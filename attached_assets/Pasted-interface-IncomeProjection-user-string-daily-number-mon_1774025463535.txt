interface IncomeProjection {
  user: string;
  daily: number;
  monthly: number;
  cumulative: number;
}

const LimitlessLiveDashboardWithIncome = () => {
  const [snapshot, setSnapshot] = useState<FullSnapshot | null>(null);
  const [income, setIncome] = useState<IncomeProjection[]>([]);

  async function fetchSnapshot() {
    try {
      const response = await fetch("http://localhost:3000/snapshot");
      const data: FullSnapshot = await response.json();
      setSnapshot(data);

      // Calculate real-time per-user projections
      const projections: IncomeProjection[] = Array(data.marketplaceUsers)
        .fill(0)
        .map((_, i) => {
          const daily = Math.round((data.totalRevenue / data.marketplaceUsers) * (Math.random() * 0.05 + 0.975));
          const monthly = daily * 30;
          const cumulative = data.totalRevenue / data.marketplaceUsers;
          return {
            user: `User-${i + 1}`,
            daily,
            monthly,
            cumulative
          };
        });
      setIncome(projections);

    } catch (err) {
      console.error("Failed to fetch snapshot:", err);
    }
  }

  useEffect(() => {
    fetchSnapshot();
    const interval = setInterval(fetchSnapshot, REFRESH_INTERVAL);
    return () => clearInterval(interval);
  }, []);

  if (!snapshot) return <div>Loading Limitless Full Snapshot + Income...</div>;

  return (
    <div style={{ padding: 20, fontFamily: "sans-serif", background: "#111", color: "#eee" }}>
      <h1 style={{ color: "#0ff" }}>🌌 Limitless Live - Full System Dashboard with Income</h1>

      <section>
        <h2>Core Metrics</h2>
        <div style={{ display: "flex", gap: "20px" }}>
          {Object.entries(snapshot.metrics).map(([k, v]) => (
            <div key={k} style={{ padding: 10, border: "1px solid #0ff", borderRadius: 6 }}>
              <strong>{k.toUpperCase()}</strong>: {v}%
            </div>
          ))}
        </div>
      </section>

      <section>
        <h2>Income Projections</h2>
        <p>Based on unlimited, fully active marketplace and emergent actions:</p>
        <table style={{ width: "100%", borderCollapse: "collapse", marginTop: 10 }}>
          <thead>
            <tr style={{ borderBottom: "1px solid #0ff" }}>
              <th>User</th>
              <th>Daily ($)</th>
              <th>Monthly ($)</th>
              <th>Cumulative ($)</th>
            </tr>
          </thead>
          <tbody>
            {income.map(i => (
              <tr key={i.user} style={{ borderBottom: "1px solid #222" }}>
                <td>{i.user}</td>
                <td>{i.daily.toLocaleString()}</td>
                <td>{i.monthly.toLocaleString()}</td>
                <td>{i.cumulative.toLocaleString()}</td>
              </tr>
            ))}
          </tbody>
        </table>
      </section>

      <section>
        <h2>Marketplace & Emergent Activity</h2>
        <p>Total Revenue Accumulated: ${snapshot.totalRevenue.toLocaleString()}</p>
        <p>Marketplace Users: {snapshot.marketplaceUsers}</p>
        <p>Emergent Modules: {snapshot.emergentModules}</p>
        <p>Dynamic Actions This Cycle: {snapshot.dynamicActions.join(", ")}</p>
      </section>

      <section>
        <h2>Universe & Upgrades</h2>
        <p>Universe Units: {snapshot.universeUnits}</p>
        <p>Meta-Phases: {snapshot.metaPhases}</p>
        <p>Expansion Ideas: {snapshot.expansionIdeas}</p>
        <p>Upgrades Registered: {snapshot.upgrades.length}</p>
      </section>

      <section>
        <h2>System Health</h2>
        <p>Server Status: {snapshot.serverStatus}</p>
        <p>Stripe Connection: {snapshot.stripeStatus}</p>
        <p>Firewall Passed: {snapshot.firewallCheck ? "Yes" : "No"}</p>
        <p>Sandbox Limit: {snapshot.sandboxLimit ? "Yes" : "No"}</p>
        <p>Gaps Fixed Automatically: {snapshot.gapsFixed ? "Yes" : "No"}</p>
        <p>Errors Fixed Automatically: {snapshot.errorsFixed ? "Yes" : "No"}</p>
      </section>
    </div>
  );
};

ReactDOM.render(<LimitlessLiveDashboardWithIncome />, document.getElementById("root"));
