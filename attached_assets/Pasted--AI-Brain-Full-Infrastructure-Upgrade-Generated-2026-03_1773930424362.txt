import { exec } from "child_process";
import { promisify } from "util";
import redis from "redis";
import { Pool } from "pg";
import { rebuildModuleShell, registerApps, validateOSContext } from "./upgrade-utils";
import { migrateMemory, setupQueueWorkers } from "./upgrade-workers";
import { runStressTest } from "./upgrade-tests";

const run = promisify(exec);

async function main() {
  console.log("🚀 Starting AI Brain Full Infrastructure Upgrade...");

  // 1. Database & Redis Setup
  console.log("1️⃣ Setting up Redis and PostgreSQL connections...");
  const redisClient = redis.createClient({ url: process.env.REDIS_URL });
  await redisClient.connect();

  const pgPool = new Pool({ connectionString: process.env.DATABASE_URL });
  console.log("✅ Redis & PostgreSQL ready");

  // 2. Migrate in-memory session & agent memory to Redis + PostgreSQL
  console.log("2️⃣ Migrating session & agent memory...");
  await migrateMemory(redisClient, pgPool);
  console.log("✅ Memory migration complete");

  // 3. Rebuild ModuleShell + AppWindow + OSContext IDs
  console.log("3️⃣ Rebuilding ModuleShell and registering apps...");
  await rebuildModuleShell();
  await registerApps();
  await validateOSContext();
  console.log("✅ ModuleShell, AppWindow, OSContext upgraded");

  // 4. Configure AI Engine Workers & Queue
  console.log("4️⃣ Setting up AI engine workers and concurrency queue...");
  await setupQueueWorkers({
    redisUrl: process.env.REDIS_URL,
    maxConcurrency: 8,
    sseTimeoutMs: 120_000,
  });
  console.log("✅ AI engines queued, SSE fixed, concurrency enabled");

  // 5. Apply security & compliance enhancements
  console.log("5️⃣ Applying security, RBAC, TLS, and GDPR/HIPAA compliance...");
  await run("npm run apply-security-upgrades"); // assume a script exists
  console.log("✅ Security & compliance enforced");

  // 6. Observability / logging
  console.log("6️⃣ Enabling logging, metrics, health checks...");
  await run("npm run enable-observability"); // assume a script exists
  console.log("✅ Observability stack online");

  // 7. Multi-node scaling & stress test
  console.log("7️⃣ Performing multi-node stress test...");
  const stressResult = await runStressTest(pgPool, redisClient);
  console.log("✅ Stress test complete:", stressResult);

  console.log("🎉 AI Brain infrastructure upgrade complete!");
  await redisClient.disconnect();
  await pgPool.end();
}

main().catch((err) => {
  console.error("❌ Upgrade failed:", err);
  process.exit(1);
});
