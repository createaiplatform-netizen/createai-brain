    <script>
        const canvas = document.getElementById('neural-canvas');
        const ctx = canvas.getContext('2d');
        const meta = document.getElementById('metadata-display');
        canvas.width = 300; canvas.height = 300;

        function generateAsset() {
            // Clear and Generate Visual Pattern
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            const seed = Math.random().toString(36).substring(7).toUpperCase();
            
            for(let i=0; i<50; i++) {
                ctx.strokeStyle = Math.random() > 0.5 ? '#d4af37' : '#ffffff';
                ctx.beginPath();
                ctx.moveTo(Math.random()*300, Math.random()*300);
                ctx.lineTo(Math.random()*300, Math.random()*300);
                ctx.stroke();
            }
            
            // Generate Metadata "Soul-Tag"
            const rarity = Math.random() > 0.9 ? 'LEGENDARY' : 'ORIGINAL';
            meta.innerHTML = `ASSET_ID: ${seed}<br>RARITY: ${rarity}<br>NEURAL_WEIGHT: ${(Math.random()*100).toFixed(2)}%`;
        }
        
        // Initial Seed
        generateAsset();
    </script>
