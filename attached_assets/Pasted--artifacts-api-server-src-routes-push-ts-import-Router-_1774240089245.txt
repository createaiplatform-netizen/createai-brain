import { Router } from "express";
import webpush from "web-push";
import { db, sql } from "../db"; // adjust for your Drizzle/SQL setup

const router = Router();

// --- VAPID Key Initialization ---
async function initVAPIDKeys() {
  // 1. Check environment variables first
  let publicKey = process.env.VAPID_PUBLIC_KEY;
  let privateKey = process.env.VAPID_PRIVATE_KEY;

  if (!publicKey || !privateKey) {
    // 2. Check DB for existing keys
    const rows = await db.execute(sql`SELECT * FROM vapid_keys LIMIT 1`);
    if (rows.length > 0) {
      publicKey = rows[0].public_key;
      privateKey = rows[0].private_key;
    } else {
      // 3. Generate new keys and store in DB
      const keys = webpush.generateVAPIDKeys();
      publicKey = keys.publicKey;
      privateKey = keys.privateKey;

      await db.execute(sql`
        INSERT INTO vapid_keys (public_key, private_key)
        VALUES (${publicKey}, ${privateKey})
      `);
    }
  }

  // 4. Configure web-push
  webpush.setVapidDetails(
    "mailto:admin@yourdomain.com", // replace with your admin/support email
    publicKey,
    privateKey
  );

  console.log("VAPID keys loaded. Public key:", publicKey);
  return { publicKey, privateKey };
}

// --- Endpoint to expose the public VAPID key ---
router.get("/vapid-key", async (req, res) => {
  try {
    const { publicKey } = await initVAPIDKeys();
    res.json({ publicKey });
  } catch (err) {
    console.error("Failed to load VAPID keys:", err);
    res.status(500).json({ error: "Failed to load VAPID keys" });
  }
});
