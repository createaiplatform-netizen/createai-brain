import crypto from 'crypto';
import { db, documents } from './artifacts/api-server/src/db'; // adjust import to match your codebase

// 1. Family contacts (example: add more as needed)
const familyContacts = [

  { name: "Nathan Stadler", email: "stadlernathan5499@gmail.com", phone: "+17157914114" },
  { name: "Carolina Stadler", email: "stadlercarolina@gmail.com", phone: "+17157914050" },
  { name: "Nolan Stadler", email: "stadlernolan29@icloud.com", phone: "+17157914050" }
];

// 2. Generate secure tokens and prepare DB entries
const messagesToInsert = familyContacts.map(contact => {
  const token = crypto.randomBytes(32).toString('hex');
  const messageContent = {
    greeting: `Hello ${contact.name}!`,
    body: "Here's your personalized message from the CreateAI Family Portal.",
    ctaLinks: {
      familyPortal: "https://createai.digital/family-hub",
      storefront: "https://createai.digital/storefront"
    }
  };

  return {
    userId: "system",
    docType: "shareable_msg",
    projectId: token,
    title: contact.name,
    body: JSON.stringify(messageContent),
    tags: token
  };
});

// 3. Insert into database
(async () => {
  try {
    for (const msg of messagesToInsert) {
      await db.insert(documents).values(msg);
      console.log(`Inserted message token for ${msg.title}: /msg/${msg.projectId}`);
