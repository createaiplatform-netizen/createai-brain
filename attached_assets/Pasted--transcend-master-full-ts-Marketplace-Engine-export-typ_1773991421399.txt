// --- Marketplace Engine ---
export type UserEarnings = {
  userId: string;
  earnings: number;
  lastAction: Date;
};

export type MarketplaceState = {
  users: Record<string, UserEarnings>;
  totalUsers: number;
  totalActions: number;
};

export const marketplaceState: MarketplaceState = {
  users: {},
  totalUsers: 0,
  totalActions: 0,
};

const BASE_EARNING = 0.75;
const USER_SHARE = 0.75;
const PLATFORM_SHARE = 0.25;
const ACTION_COOLDOWN = 1000 * 60 * 60; // 1 hour

export function registerUser(userId: string) {
  if (!marketplaceState.users[userId]) {
    marketplaceState.users[userId] = { userId, earnings: 0, lastAction: new Date(0) };
    marketplaceState.totalUsers++;
  }
}

export function recordAction(userId: string, actionType: string): number {
  const user = marketplaceState.users[userId];
  if (!user) throw new Error("User not registered");

  const now = new Date();
  const elapsed = now.getTime() - user.lastAction.getTime();
  if (elapsed < ACTION_COOLDOWN) return 0;

  const activeUsersFactor = 1 + marketplaceState.totalUsers * 0.02;
  const totalActionsFactor = 1 + marketplaceState.totalActions * 0.01;
  const scaledEarning = BASE_EARNING * activeUsersFactor * totalActionsFactor;

  const userEarning = scaledEarning * USER_SHARE;
  user.earnings += userEarning;

  marketplaceState.totalActions++;
  user.lastAction = now;

  return userEarning;
}

export function runMarketplaceDemo() {
  const perUser = Object.values(marketplaceState.users).map(u => ({
    userId: u.userId,
    earnings: u.earnings.toFixed(2),
  }));

  const avgEarnings =
    perUser.reduce((sum, u) => sum + parseFloat(u.earnings), 0) / perUser.length || 0;

  return {
    perUser,
    scaledTotal: (avgEarnings * 1_000_000).toFixed(2),
  };
}
