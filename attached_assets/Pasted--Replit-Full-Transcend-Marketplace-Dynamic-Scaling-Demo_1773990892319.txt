// ---------------------------
// HELPER: DYNAMIC EARNINGS CALCULATION
// ---------------------------
function updateEarnings(userId, baseValue) {
  const user = users.find(u => u.id === userId);
  if (!user) return;

  // Scaling factor: increases as more users and items participate
  const activeUsersFactor = 1 + users.length * 0.02; // +2% per user
  const itemsSoldFactor = 1 + marketplace.reduce((acc, i) => acc + i.sold, 0) * 0.01; // +1% per item sold

  const scaledValue = baseValue * activeUsersFactor * itemsSoldFactor;

  const userShare = scaledValue * 0.75;
  const platformShare = scaledValue * 0.25;

  user.earnings += userShare;

  console.log(
    `User ${user.name} earned $${userShare.toFixed(2)} | Platform earned $${platformShare.toFixed(2)} (Scaled)`
  );
}

// ---------------------------
// USER ACTIONS
// ---------------------------
function clickTranscend(userId) {
  const baseValue = 1;
  updateEarnings(userId, baseValue);
  console.log(`Transcend clicked by ${users.find(u => u.id === userId).name}`);
}

function completeModule(userId, moduleName) {
  const module = modules.find(m => m.name === moduleName);
  if (!module) return;
  const baseValue = module.score * module.overachievement_pct / 100;
  updateEarnings(userId, baseValue);
  console.log(`Module ${moduleName} completed by ${users.find(u => u.id === userId).name}`);
}

// ---------------------------
// MARKETPLACE
// ---------------------------
function createItem(userId, title, price) {
  const id = marketplace.length + 1;
  marketplace.push({ id, creatorId: userId, title, price, sold: 0 });
  console.log(`Item "${title}" created by ${users.find(u => u.id === userId).name} | Price: $${price}`);
}

function buyItem(buyerId, itemId) {
  const item = marketplace.find(i => i.id === itemId);
  if (!item) return;
  item.sold += 1;
  updateEarnings(item.creatorId, item.price);
  console.log(`${users.find(u => u.id === buyerId).name} bought "${item.title}" | Price: $${item.price}`);
}
