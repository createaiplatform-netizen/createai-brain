        <div class="control-grid">
            <button class="btn" onclick="processSeed()">MANUAL_STRIKE</button>

    <script>
        const canvas = document.getElementById('neural-canvas');
        const ctx = canvas.getContext('2d');
        const vaultList = document.getElementById('vault-list');
        const status = document.getElementById('status');
        const autoBtn = document.getElementById('auto-btn');
        
        let isAuto = false;
        let autoInterval;

        function toggleAuto() {
            isAuto = !isAuto;
            if (isAuto) {
                autoBtn.innerText = "AUTO_SCAN: ON";
                autoBtn.classList.add('active');
                autoInterval = setInterval(processSeed, 150); // High-speed scanning
            } else {
                autoBtn.innerText = "AUTO_SCAN: OFF";
                autoBtn.classList.remove('active');
                clearInterval(autoInterval);
                status.innerText = "SYSTEM_IDLE...";
                status.style.color = "white";
            }
        }

        function processSeed() {
            ctx.clearRect(0, 0, 300, 300);
            const id = "Z-" + Math.random().toString(36).substring(7).toUpperCase();
            const isLegendary = Math.random() > 0.96; // Only 4% are Legendary
            
            ctx.strokeStyle = isLegendary ? '#d4af37' : '#222';
            ctx.lineWidth = isLegendary ? 4 : 0.5;
            
            for(let i=0; i < (isLegendary ? 80 : 20); i++) {
                ctx.beginPath();
                ctx.moveTo(Math.random()*300, Math.random()*300);
                ctx.lineTo(Math.random()*300, Math.random()*300);
                ctx.stroke();
            }

            if(isLegendary) {
                status.innerText = `LEGENDARY_FOUND: ${id}`;
                status.style.color = "lawngreen";
                addToVault(id);
            } else if (!isAuto) {
                status.innerText = "DENSITY_LOW: REJECTED";
                status.style.color = "white";
            }
        }

        function addToVault(id) {
            const dataURL = canvas.toDataURL("image/png");
            const item = document.createElement('div');
            item.className = 'vault-item';
            item.innerHTML = `<span>${id}</span><a class="export-link" download="${id}.png" href="${dataURL}">EXPORT</a>`;
            vaultList.prepend(item);
        }
    </script>
