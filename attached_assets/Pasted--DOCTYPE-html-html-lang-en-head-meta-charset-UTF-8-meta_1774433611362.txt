    <script>
        const canvas = document.getElementById('neural-canvas');
        const ctx = canvas.getContext('2d');
        const vaultList = document.getElementById('vault-list');
        const status = document.getElementById('status');

        function processSeed() {
            ctx.clearRect(0, 0, 300, 300);
            const id = Math.random().toString(36).substring(7).toUpperCase();
            const isLegendary = Math.random() > 0.85; // 15% chance
            
            // Draw Neural Pattern
            ctx.strokeStyle = isLegendary ? '#d4af37' : '#444';
            ctx.lineWidth = isLegendary ? 2 : 1;
            for(let i=0; i<40; i++) {
                ctx.beginPath();
                ctx.moveTo(Math.random()*300, Math.random()*300);
                ctx.lineTo(Math.random()*300, Math.random()*300);
                ctx.stroke();
            }

            if(isLegendary) {
                status.innerText = `LEGENDARY_ASSET_DETECTED: ${id}`;
                status.style.color = "lawngreen";
                addToVault(id);
            } else {
                status.innerText = "ASSET_REJECTED: LOW_DENSITY";
                status.style.color = "white";
            }
        }

        function addToVault(id) {
            const time = new Date().toLocaleTimeString();
            if(vaultList.innerHTML.includes('EMPTY_VAULT')) vaultList.innerHTML = '';
            const item = document.createElement('div');
            item.className = 'vault-item';
            item.innerHTML = `<span>ID: ${id}</span> <span>[SECURED_${time}]</span>`;
            vaultList.prepend(item);
        }
    </script>
