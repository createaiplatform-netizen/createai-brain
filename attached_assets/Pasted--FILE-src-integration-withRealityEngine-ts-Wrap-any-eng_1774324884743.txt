// =========================
// FILE: src/integration/withRealityEngine.ts
// Wrap any engine so it always runs under the Absolute Layer
// =========================
import { orchestratorLayer } from "../reality/orchestratorLayer";

export type Engine<I, O> = (input: I, context?: unknown) => Promise<O> | O;

export function withRealityEngine<I, O>(engine: Engine<I, O>): Engine<I, O> {
  return (input: I, context?: unknown) => {
    const wrapped = orchestratorLayer.applyAbsolute({ input, context });
    return engine(wrapped as unknown as I, context);
  };
}

// =========================
// FILE: src/integration/apiMiddleware.ts
// Minimal API helpers that can be used where you want, not auto‑wired
// =========================
import type { Request, Response, NextFunction } from "express";
import { orchestratorLayer } from "../reality/orchestratorLayer";
import { metaCosmic } from "../meta/metaCosmic";

export function absoluteLayerMiddleware(req: Request, _res: Response, next: NextFunction) {
  (req as any).absolute = orchestratorLayer.applyAbsolute({ path: req.path, method: req.method });
  next();
}

export function metaAlignmentMiddleware(req: Request, res: Response, next: NextFunction) {
  const alignment = metaCosmic.alignmentEngine().check({ path: req.path, method: req.method });
  if (!alignment.alignedWithPurpose) {
    return res.status(403).json({ error: "Request not aligned with system purpose." });
  }
  next();
}

// FILE: src/expansion/emotionalFabric.ts
// Lightweight expansion layer that plugs into metaCosmic + realityStack
// =========================
import { metaCosmic } from "../meta/metaCosmic";

export type EmotionalState = "calm" | "stressed" | "curious" | "overwhelmed";

export const emotionalFabric = {
  interpret(state: EmotionalState) {
    const purpose = metaCosmic.purposeEngine();
    return {
      state,
      supportMode: state === "overwhelmed" ? "stabilize" : "gently_expand",
      reminder: purpose.corePurpose
    };
  }
};

// FILE: src/expansion/temporalLattice.ts
// Minimal temporal layer for continuity hooks
// =========================
export type TimelinePoint = {
  id: string;
  label: string;
  timestamp: number;
};

export const temporalLattice = {
  createPoint(label: string): TimelinePoint {
    return {
      id: `${Date.now()}-${Math.random().toString(16).slice(2)}`,
      label,
      timestamp: Date.now()
    };
  }
};

// Minimal family universe layer (you can grow this later)
// =========================
import type { IdentityEssence } from "./identityField";

export type FamilyMember = IdentityEssence & {
  relation: "self" | "partner" | "child" | "other";
};

export const familyUniverse = {
  members: [] as FamilyMember[],
  add(member: FamilyMember) {
    this.members.push(member);
  },
  list() {
    return this.members;
  }
};
