// ==== SEND EMAILS & SMS ====
async function sendMessages() {
  for (const contact of contacts) {
    try {
      // Send email via Resend
      await resend.emails.send({
        from: "sara@createai.digital",
        to: contact.email,
        subject: contact.emailSubject,
        text: contact.emailBody,
        html: `<pre>${contact.emailBody}</pre>`
      });
      console.log(`✅ Email sent to ${contact.name} (${contact.email})`);
    } catch (err) {
      console.error(`❌ Email failed for ${contact.name}:`, err);
    }

    try {
      // Send SMS via Twilio
      await twilioClient.messages.create({
        from: twilioFrom,
        to: contact.phone,
        body: contact.smsBody
      });
      console.log(`✅ SMS sent to ${contact.name} (${contact.phone})`);
    } catch (err) {
      console.error(`❌ SMS failed for ${contact.name}:`, err);
    }
  }
}

// Execute
sendMessages()
  .then(() => console.log("All messages processed"))
  .catch(err => console.error("Error sending messages:", err));
