// File: UltimateRenderEngineApp.tsx
import React, { useState, useEffect } from "react";
import { generateProjectAssets } from "./RenderEngineBackend";
import { ProjectType } from "./ProjectTypes";

interface UltimateRenderEngineAppProps {
  projectId: string;
  projectName: string;
  projectType: ProjectType;
  onClose: () => void;
}

// Unified render engine handling all project types
export const UltimateRenderEngineApp: React.FC<UltimateRenderEngineAppProps> = ({
  projectId,
  projectName,
  projectType,
  onClose,
}) => {
  const [progress, setProgress] = useState<string[]>([]);
  const [outputURL, setOutputURL] = useState<string | null>(null);
  const [isGenerating, setIsGenerating] = useState(false);
  const [stepNumber, setStepNumber] = useState(0);

  // Main generation pipeline
  const startGeneration = async () => {
    setIsGenerating(true);
    const streamCallback = (msg: string) => {
      setStepNumber(prev => prev + 1);
      setProgress(prev => [...prev, `Step ${stepNumber + 1}: ${msg}`]);
    };

    // Backend call: reads full project, generates visual/audio/interactive assets
    const resultURL = await generateProjectAssets(projectId, projectType, streamCallback);

    setOutputURL(resultURL);
    setIsGenerating(false);
  };

  // Auto-start generation
  useEffect(() => {
    startGeneration();
  }, []);

  // Render output player based on project type
  const renderOutput = () => {
    switch (projectType) {
      case "Film":
        return <video src={outputURL!} controls autoPlay style={{ width: "100%" }} />;
      case "Game":
      case "App":
        return <iframe src={outputURL!} sandbox="allow-scripts allow-same-origin" style={{ width: "100%", height: "600px" }} />;
      case "Book":
      case "Course":
      case "Pitch":
      case "PhysicalProduct":
        return <iframe src={outputURL!} style={{ width: "100%", height: "600px" }} />;
      case "Music":
      case "Podcast":
        return <audio src={outputURL!} controls autoPlay style={{ width: "100%" }} />;
      default:
        return <div>Unsupported project type</div>;
    }
  };

  return (
    <div className="render-overlay" style={{
      position: "fixed",
      top: 0, left: 0, right: 0, bottom: 0,
      backgroundColor: "rgba(0,0,0,0.85)",
      color: "#fff",
      overflowY: "auto",
      padding: "20px",
      zIndex: 9999
    }}>
      <button 
        className="close-btn" 
        onClick={onClose} 
        style={{
          position: "absolute",
          top: 10, right: 10,
          fontSize: "24px",
          background: "none",
          border: "none",
          color: "#fff",
          cursor: "pointer"
        }}>×</button>

      <h2 style={{ marginBottom: "10px" }}>Generating: {projectName}</h2>

      <div className="progress-log" style={{
        maxHeight: "200px",
        overflowY: "scroll",
        border: "1px solid #444",
        padding: "10px",
        marginBottom: "20px"
      }}>
        {progress.map((p, i) => <div key={i}>{p}</div>)}
        {isGenerating && <div>Processing...</div>}
      </div>

      {outputURL && (
        <div className="output-container" style={{ marginTop: "20px" }}>
          {renderOutput()}
        </div>
      )}
    </div>
  );
};
