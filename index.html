// THE HUSBAND'S SENSOR SWEEP function draw() { ctx.fillStyle = 'rgba(0,0,0,0.2)'; ctx.fillRect(0,0,canvas.width, canvas.height); // The 'Radar' Pulse if(touch.x > 0) { ctx.strokeStyle = 'rgba(34, 139, 34, 0.3)'; // Tactical Green Pulse ctx.lineWidth = 2; ctx.beginPath(); ctx.arc(touch.x, touch.y, (Date.now() % 1000) / 5, 0, Math.PI * 2); ctx.stroke(); } lines.forEach(s => { let dx = touch.x - s.x; let dy = touch.y - s.y; let dist = Math.sqrt(dx*dx + dy*dy); // Bend light toward touch if(dist < 150) { s.oX += (dx * 0.08); ctx.strokeStyle = '#fff'; // Turn white when touched (Tactical Highlight) } else { s.oX *= 0.92; ctx.strokeStyle = '#d4af37'; // Stay gold in the shadows } ctx.beginPath(); ctx.moveTo(s.x + s.oX, s.y); ctx.lineTo(s.x + (s.oX * 0.5), s.y + s.l); ctx.stroke(); s.y += s.v; if(s.y > canvas.height) { s.y = -s.l; s.x = Math.random()*canvas.width; } }); requestAnimationFrame(draw); }
