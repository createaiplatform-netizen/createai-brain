// ADD TO SCRIPT SECTION let systemLogs = []; function act(name, color, sector) { tint = color; // Create Log Entry const timestamp = new Date().toLocaleTimeString(); const entry = `[${timestamp}] ACCESS: ${name} (${sector})`; systemLogs.unshift(entry); // Add newest to top // Display UI document.getElementById('overlay').style.display = 'flex'; document.getElementById('ot').innerText = name; document.getElementById('ot').style.color = color; document.getElementById('os').innerText = sector; // Console output for Husband to monitor console.log("%c" + entry, "color: " + color + "; font-weight: bold;"); } // Add this function to view the full log history function viewLogs() { alert("SYSTEM LOG HISTORY:\n\n" + systemLogs.slice(0, 5).join("\n")); }
